Calculate Average Grades

def calculate_average():
    students = {}
    while True:
        name = input("Enter student's name (or 'exit' to stop): ")
        if name.lower() == 'exit':
            break

        try:
            grade = float(input(f"Enter grade for {name}: "))
            students[name] = grade
        except ValueError:
            print("Invalid input. Please enter a numeric grade.")
            continue

    avg_grade = sum(students.values()) / len(students) if students else 0
    print(f"Average grade: {avg_grade:.2f}")

    <?php
    function factorial($n) {
        if ($n < 0) {
            return "Factorial is not defined for negative numbers.";
        }
        $fact = 1;
        for ($i = 1; $i <= $n; $i++) {
            $fact *= $i;
        }
        return $fact;
    }

    if ($_SERVER["REQUEST_METHOD"] == "POST") {
        $num = intval($_POST["num"]);
        echo "<h3>Factorial of $num is " . factorial($num) . "</h3>";
    }
    ?>

Flask Login Auth
Python/Flask 
from flask import Flask, request, render_template_string

app = Flask(__name__)

users = {'admin': 'password123'}

@app.route('/', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        username = request.form['username']
        password = request.form['password']
        if users.get(username) == password:
            return 'Login successful!'
        else:
            return 'Invalid credentials.'

    return render_template_string('''
        <form method="POST">
            Username: <input type="text" name="username"><br>
            Password: <input type="password" name="password"><br>
            <input type="submit" value="Login">
        </form>
    ''')

if __name__ == '__main__':
    app.run(debug=True)

            <option value="palindrome">Check if palindrome</option>
        </select><br><br>
        <input type="submit" value="Perform Operation">
    </form>

    <?php
    function isPalindrome($str) {
        $reversed = strrev($str);
        return strtolower($str) == strtolower($reversed);
    }

    if ($_SERVER["REQUEST_METHOD"] == "POST") {
        $string = $_POST["string"];
        $operation = $_POST["operation"];
        echo "<h3>Result:</h3>";

        switch ($operation) {
            case "reverse":
                echo "Reversed string: " . strrev($string);
                break;
            case "uppercase":
                echo "Uppercase string: " . strtoupper($string);
                break;
            case "lowercase":
                echo "Lowercase string: " . strtolower($string);
                break;
            case "length":
                echo "Length of the string: " . strlen($string);
                break;
            case "palindrome":
                echo isPalindrome($string) ? "The string is a palindrome." : "The string is not a palindrome.";
                break;
            default:
                echo "Invalid operation.";
        }
    }
    ?>
</body>

const express = require('express');
const app = express();
const PORT = 3000;

// Route to handle GET request to /hello
app.get('/hello', (req, res) => {
  res.send('Hello, World!');
});

// Fallback route for anything else (404)
app.use((req, res) => {
  res.status(404).send('404 Not Found - The requested resource does not exist.');
});

// Start the server
app.listen(PORT, () => {
  console.log(`Server is running on http://localhost:${PORT}`);
});

Flask Multi-Page App
Python/Flask 
from flask import Flask, render_template_string
import random

app = Flask(__name__)

quotes = [
    "The best way to predict the future is to invent it.",
    "Life is 10% what happens to us and 90% how we react to it.",
    "You miss 100% of the shots you don't take."
]

@app.route('/')
def welcome():
    return render_template_string('''
        <h1>Welcome to My Flask App!</h1>
        <a href="/quote">Go to the Quote Page</a>
    ''')

@app.route('/quote')
def show_quote():
    quote = random.choice(quotes)
    return render_template_string(f'''
        <h1>Random Quote:</h1>
        <p>"{quote}"</p>
        <a href="/">Back to Home Page</a>
    ''')

if __name__ == '__main__':
    app.run(debug=True)
